### 1.2 RAG Prompt 模板——最关键的一环

Prompt 怎么写直接影响 RAG 的质量。一个经典的 RAG Prompt 模板：

```
你是一个基于文档库回答问题的 AI 助手。

## 规则
1. 只根据下面提供的「参考资料」回答问题
2. 如果参考资料中没有相关信息，直接说「根据现有资料无法回答」
3. 回答时标注引用的文档编号，例如 [来源: doc_3]
4. 不要编造参考资料中没有的信息

## 参考资料
{这里插入检索到的文档块}

## 用户问题
{用户的问题}

## 你的回答
```

<TipBox title="Prompt 设计对 RAG 至关重要" icon={Lightbulb}>
  一个好的 RAG Prompt 要做三件事：
  1. **约束**：告诉 LLM 只能基于参考资料回答
  2. **诚实**：没有相关信息时允许它说「不知道」
  3. **可追溯**：要求标注来源，让用户能验证
</TipBox>

## 2. 动手实践：构建完整 RAG 系统

### 完整代码

创建 `rag_app.py`：

```python
import streamlit as st
import os
import chromadb
from chromadb.utils import embedding_functions
from anthropic import Anthropic
from dotenv import load_dotenv
import PyPDF2  # pip install PyPDF2
import io

load_dotenv()

# ---- 页面配置 ----
st.set_page_config(page_title="📚 文档问答 AI", page_icon="📚")
st.title("📚 让 AI 读你的文档")

# ---- 初始化 ----
client = Anthropic(api_key=os.getenv("CLAUDE_API_KEY"))
chroma_client = chromadb.PersistentClient(path="./rag_db")

@st.cache_resource
def get_collection():
    return chroma_client.get_or_create_collection(
        name="user_docs",
        embedding_function=embedding_functions.OpenAIEmbeddingFunction(
            api_key=os.getenv("OPENAI_API_KEY"),
            model_name="text-embedding-3-small"
        )
    )

collection = get_collection()

# ---- 文档上传 ----
st.sidebar.header("📄 上传文档")
uploaded_file = st.sidebar.file_uploader(
    "支持 PDF 和 TXT",
    type=["pdf", "txt"],
    accept_multiple_files=False
)

def chunk_text(text: str, chunk_size: int = 500, overlap: int = 50) -> list[str]:
    chunks = []
    start = 0
    while start < len(text):
        end = min(start + chunk_size, len(text))
        if end < len(text):
            for sep in ["。\n", "。", "\n", "，"]:
                last_sep = text.rfind(sep, start, end)
                if last_sep > start + chunk_size * 0.5:
                    end = last_sep + 1
                    break
        chunk = text[start:end].strip()
        if chunk:
            chunks.append(chunk)
        start = end - overlap
    return chunks

if uploaded_file:
    # 读取文件
    if uploaded_file.type == "application/pdf":
        reader = PyPDF2.PdfReader(io.BytesIO(uploaded_file.read()))
        text = ""
        for page in reader.pages:
            text += page.extract_text() + "\n"
    else:
        text = uploaded_file.read().decode("utf-8")

    st.sidebar.success(f"✅ 已读取 {len(text)} 个字符")

    # 分块
    chunks = chunk_text(text, chunk_size=500, overlap=50)
    st.sidebar.info(f"✂️ 切成 {len(chunks)} 个块")

    # 存入向量数据库
    if st.sidebar.button("📥 存入知识库"):
        with st.spinner("正在向量化文档..."):
            collection.add(
                documents=chunks,
                ids=[f"chunk_{i}" for i in range(len(chunks))],
            )
        st.sidebar.success(f"✅ 已存入 {len(chunks)} 个文档块！")

# ---- 问答区域 ----
st.header("💬 基于文档提问")

query = st.text_input("输入你的问题：", placeholder="例如：这份文档主要讲了什么？")

if query:
    # Step 1: 检索
    with st.spinner("🔍 正在检索相关文档..."):
        results = collection.query(
            query_texts=[query],
            n_results=3,
        )

        retrieved_docs = results["documents"][0]
        doc_ids = results["ids"][0]

        # 显示检索结果
        with st.expander(f"📎 检索到 {len(retrieved_docs)} 个相关文档块", expanded=False):
            for i, (doc_id, doc_text) in enumerate(zip(doc_ids, retrieved_docs)):
                st.caption(f"**[来源: {doc_id}]** (相关度 #{i+1})")
                st.text(doc_text[:300] + "...")

    # Step 2&3: 增强 + 生成
    with st.spinner("🤔 AI 正在基于文档思考..."):
        # 构建 RAG Prompt
        context = "\n\n---\n\n".join([
            f"[来源: {doc_id}]\n{doc_text}"
            for doc_id, doc_text in zip(doc_ids, retrieved_docs)
        ])

        rag_prompt = f"""你是一个基于文档库回答问题的 AI 助手。

## 规则
1. 只根据下面提供的「参考资料」回答问题
2. 如果参考资料中没有相关信息，直接说「根据现有资料无法回答」
3. 回答时标注引用的来源编号
4. 不要编造参考资料中没有的信息

## 参考资料
{context}

## 用户问题
{query}

## 你的回答"""

        # 调用 LLM
        response = client.messages.create(
            model="claude-haiku-4-5-20251001",
            max_tokens=500,
            system="你是一个严谨的 AI 助手。你只基于提供的参考资料回答问题，绝不编造。",
            messages=[{"role": "user", "content": rag_prompt}]
        )

        ai_reply = response.content[0].text

        # 显示回答
        st.subheader("🤖 AI 回答")
        st.markdown(ai_reply)

        # 显示 token 使用
        usage = response.usage
        st.caption(
            f"📊 检索到 {len(retrieved_docs)} 个文档块 · "
            f"输入 {usage.input_tokens} tokens · "
            f"输出 {usage.output_tokens} tokens"
        )

# ---- 统计信息 ----
st.sidebar.divider()
st.sidebar.caption(f"📚 知识库文档块总数：{collection.count()}")
```

<WarningBox title="常见踩坑">
  - **PDF 中文乱码**：PyPDF2 对中文字体支持不完美。考虑用 `pdfplumber` 替代，或者直接用 TXT 格式。
  - **OpenAI Embedding API 需要单独申请**：不同于 ChatGPT API，Embedding API 也需要在 OpenAI 后台开通。
  - **检索结果不相关**：试试调整 Chunk Size（更小 = 更精准但缺上下文），或者增加 `n_results` 参数。
  - **回答太长超出上下文**：控制检索返回的文档块数量和每块大小，确保总 token 数在模型上下文窗口内。
</WarningBox>
